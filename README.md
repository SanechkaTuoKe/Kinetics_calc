Адсорбция
C = A / 27905.64 × разбавление
С₀(адсорбированное) = С_нач_фур_в_растворе - C
% адсорбции = (С₀ / С_нач) × 100%

Высвобождение
C = (A + 0.00608) / 27905.64
C_6мл = C × 6
Сумма_C = Σ(C_6мл_i) для i = 1 до k
% высвобождения = (Сумма_C / С₀) × 100%
